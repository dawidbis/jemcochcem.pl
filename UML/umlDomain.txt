    class MacroNutrients <<record>> {
        + decimal Protein

    }

package "DomainInterfaces" {
    interface INutritionCalculationService {
        + CalculateItemCalories(grams: decimal, caloriesPer100g: int): int
        + CalculateItemMacros(grams: decimal, proteinPer100g: decimal, carbs: decimal, fats: decimal): MacroNutrients
    }

    interface IMealLogDomainService {
        + AddItemToLog(log: MealLog, item: MealLogItem): void
        + RecalculateLogTotals(log: MealLog): void
    }
}

package "DomainServices" {
    class NutritionCalculationService implements INutritionCalculationService {
        + CalculateItemCalories(grams: decimal, caloriesPer100g: int): int
        + CalculateItemMacros(grams: decimal, proteinPer100g: decimal, carbs: decimal, fats: decimal): MacroNutrients
    }

    class TdeeCalculationService {
        + CalculateBMR(weight: decimal, height: decimal, age: int, gender: string): decimal
        + CalculateTDEE(weight: decimal, activityLevel: decimal): int
    }

    class MealLogDomainService implements IMealLogDomainService {
        - INutritionCalculationService _nutritionService
        + AddItemToLog(log: MealLog, item: MealLogItem): void
        + RecalculateLogTotals(log: MealLog): void
    }
}
